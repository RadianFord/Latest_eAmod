// -------------- Criador de Pecopecos (para Cavaleiros e evoluções) --------------
prontera,55,350,5	script	Criador de Pecopecos#cav	105,{
	if(Upper == 0 || Upper == 1 || Upper == 2) set .@price,2500; // Pecopeco Normal - Avançado e Bebê

	mes "[Criador de Pecopecos]";
	if (BaseJob == Job_Knight && Class < Job_Rune_Knight) {
		mes "Bom-dia honorável Cavaleiro";
		mes "gostaria de comprar";
		mes "um Pecopeco? Você precisa ter a Habilidade";
		mes "Montaria e apenas "+.@price+" zeny sobrando.";
		next;
		switch(select("Alugar Peco Peco:Cancelar")) {
		case 1:
			if (zeny < .@price) {
				mes "[Criador de Pecopecos]";
				mes "Você não possui";
				mes "zeny suficiente.";
				mes "Você está...";
				mes "pobre?";
				close;
			} else if(getskilllv("KN_RIDING") == 0) {
				mes "[Criador de Pecopecos]";
				mes "Me desculpe, mas você";
				mes "não pode utilizar esse";
				mes "serviço. Por favor aprenda";
				mes "a habilidade Montaria primeiro.";
				close;
			} else if(checkriding() || ismounting()) {
				mes "[Criador de Pecopecos]";
				mes "Você já está";
				mes "montado em uma montaria.";
				close;
			}	
			set zeny,zeny-.@price;
			setriding;
			close;
		case 2:
			mes "[Criador de Pecopecos]";
			mes "Entendo.";
			mes "Bem então,";
			mes "tenha um bom dia.";
			close;
		}
	} else {
		mes "Me desculpe, mas esses";
		mes "Peco Pecos estão disponíveis";
		mes "apenas para Cavaleiros";
		mes "e Lordes.";
		close;
	}
}

			setdragon;
			close;
		case 2:
			mes "[Criador de Dragões]";
			mes "Entendo. Então tenha um bom dia.";
			close;
		}
	}
	mes "O que você está fazendo aqui?";
	mes "Apenas Cavaleiros Rúnicos podem obter um Dragão.";
	close;
}

// -------------- NPC Removedor de Pecopecos --------------

prontera,125,208,5	script#re	Removedor de Pecopecos	105,{
	mes "[Soldado]";
	mes "Se voce deseja desmontar de seu Pecopeco";
	mes "por alguma razão, por favor";
	mes "fique a vontade para utilizar meu serviço.";
	if (checkriding()) {
		next;
		mes "[Soldado]";
		mes "Você está montado em um Pecopeco.";
		mes "Deseja removê-lo?";
		next;
		switch(select("Sim:Não")) {
		case 1:
			setriding 0;
			mes "[Soldado]";
			mes "Me diga, como se sente";
			mes "ao tocar o chão novamente?";
			close;
		case 2:
			mes "[Soldado]";
			mes "Entendo. Por favor se mudar de idéia";
			mes "fique a vontade para voltar quando quiser.";
			close;
		}
	}
	close;
}
